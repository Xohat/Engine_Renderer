30 // Carga de meshes desde un archivo (Autor TheCherno)
31 static shader_code parse_shader(const::std::string& path)
32 {
33 // Carga el archivo
34 std::ifstream stream(path);
35
36 // Tipo de shader (Tener en cuenta los valores del enum)
37 enum class shader_type
38 {
39 NONE = -1,
40 VERTEX = 0,
41 FRAGMENT = 1
42 };
43
44 // Línea a analizar ahora
45 std::string line;
46
47 // Array que se va a a devolver
48 // Hueco 0: Vertex shader
49 // Hueco 1: Fragment shader
50 std::stringstream return_stream[2];
51 shader_type type = shader_type::NONE;
52
53 // Revisa todo el documento viendo todo el archivo por stream y leyendo línea por línea
54 while (getline(stream, line))
55 {
56 // Por esto la sintaxis de los shaders debe ser
57 // #shader vertex
58 // #shader fragment
59 // Ya que primero encuentra las líneas que tengan shader y tras eso la palabra clave vertex o fragment
60 if (line.find("#shader") != std::string::npos)
61 {
62 // Si encuentra una línea tras vertex la introduce en el
63 // hueco de shader_type::VERTEX que al ser un enum tiene valor de 0
64 if (line.find("vertex") != std::string::npos)
65 {
66 type = shader_type::VERTEX;
67 }
68 // Si encuentra una línea tras fragment la introduce en el
69 // hueco de shader_type::fragment que al ser un enum tiene valor de 1
70 else if (line.find("fragment") != std::string::npos)
71 {
72 type = shader_type::FRAGMENT;
73 }
74 }
75 else
76 {
77 // Si la línea no tiene ninguna de las dos marcas se supone que es el código
78 // correspondietne antes de entrar en el siguiente fragment maracdo por el line.find()
79 return_stream[(int)type] << line << '\n';
80 }
81 }
82
83 // Tras acabar de cargar el array se devuelven los datos en el formato de un struct shader_code
84 return { return_stream[0].str(), return_stream[1].str() };
85 }

112 GLuint compile_shaders()

114 // Carga el shader desde el archivo
115 source = parse_shader("../../code/resources/shader/BasicShader.shader");

117 GLint succeeded = GL_FALSE;

119 // Se crean objetos para los shaders:
120
121 GLuint vertex_shader_id = glCreateShader(GL_VERTEX_SHADER);
122 GLuint fragment_shader_id = glCreateShader(GL_FRAGMENT_SHADER);
123
124 const char* vertex_shaders_code[] = { source.vertex_code.c_str() };
125 const char* fragment_shaders_code[] = { source.fragment_code.c_str() };
126 const GLint vertex_shaders_size[] = { (GLint)source.vertex_code.size() };
127 const GLint fragment_shaders_size[] = { (GLint)source.fragment_code.size() };
128
129 glShaderSource(vertex_shader_id, 1, vertex_shaders_code, vertex_shaders_size);
130 glShaderSource(fragment_shader_id, 1, fragment_shaders_code, fragment_shaders_size);
131
132 // Se compilan los shaders:
133
134 glCompileShader(vertex_shader_id);
135 glCompileShader(fragment_shader_id);
136
137 // Se comprueba que si la compilación ha tenido éxito:
138
139 glGetShaderiv(vertex_shader_id, GL_COMPILE_STATUS, &succeeded);
140 if (!succeeded) debugger.show_compilation_error(vertex_shader_id);
141
142 glGetShaderiv(fragment_shader_id, GL_COMPILE_STATUS, &succeeded);
143 if (!succeeded) debugger.show_compilation_error(fragment_shader_id);
144
145 // Se crea un objeto para un programa:
146
147 GLuint program_id = glCreateProgram();
148
149 // Se cargan los shaders compilados en el programa:
150
151 glAttachShader(program_id, vertex_shader_id);
152 glAttachShader(program_id, fragment_shader_id);
153
154 // Se linkan los shaders:
155
156 glLinkProgram(program_id);
157
158 // Se comprueba si el linkage ha tenido éxito:
159
160 glGetProgramiv(program_id, GL_LINK_STATUS, &succeeded);
161 if (!succeeded) debugger.show_linkage_error(program_id);
162
163 // Se liberan los shaders compilados una vez se han linkado:
164
165 glDeleteShader(vertex_shader_id);
166 glDeleteShader(fragment_shader_id);
167
168 current_program_id = program_id;
169
170 return (program_id);
171 }
